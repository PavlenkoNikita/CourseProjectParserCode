private void CheckEndBlockConstructionNoBracket()
        {
            bool checkEndConstruction = false;
            switch (callsConstructions.Peek())
            {
                case "FOR":
                    {
                        checkEndConstruction = true;
                        callCycle.Pop();
						return;
                        break;
                    }
                case "IF":
                    {
                        checkEndConstruction = true;
                        break;
                    }
                case "WHILE":
                    {
                        checkEndConstruction = true;
                        callCycle.Pop();
                        break;
                    }
                case "ELSE":
                    {
                        checkEndConstruction = true;
                        break;
                    }
            }
            
        }}
